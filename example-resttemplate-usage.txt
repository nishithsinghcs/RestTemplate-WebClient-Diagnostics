RestTemplateDiagnostics diagnostics = new RestTemplateDiagnostics(restTemplate);

ResponseWrapper<String> result =
        diagnostics.callWithDiagnostics("http://internal-service/api/data", String.class);

if (result.isSuccess()) {
    log.info("✅ Got body={} in {}ms", result.getBody(), result.getExecutionTimeMs());
} else {
    log.warn("⚠️ Failed with status={} error={} time={}ms", 
             result.getStatus(), result.getErrorMessage(), result.getExecutionTimeMs());
}
