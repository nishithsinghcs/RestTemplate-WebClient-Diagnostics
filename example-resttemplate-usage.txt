RestTemplateDiagnostics diagnostics = new RestTemplateDiagnostics(restTemplate);

// Expect full response
ResponseEntity<String> response = diagnostics.callWithDiagnostics("http://internal-service/api/data", String.class);

if (response.getStatusCode().is2xxSuccessful()) {
    log.info("✅ Body: {}", response.getBody());
} else {
    log.warn("⚠️ Failed with status: {}", response.getStatusCode());
}
