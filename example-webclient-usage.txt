WebClient webClient = WebClient.builder().baseUrl("http://service1").build();
WebClientDiagnostics diagnostics = new WebClientDiagnostics(webClient);

// Call with tag for traceability
ResponseWrapper<String> resp =
        diagnostics.callWithDiagnostics("/api/data", String.class, "MS1");

if (resp.isSuccess()) {
    log.info("✅ [{}] Got response body={} in {}ms", resp.getTag(), resp.getBody(), resp.getExecutionTimeMs());
} else {
    log.warn("⚠️ [{}] Failed with status={} error={} time={}ms",
             resp.getTag(), resp.getStatus(), resp.getErrorMessage(), resp.getExecutionTimeMs());
}
